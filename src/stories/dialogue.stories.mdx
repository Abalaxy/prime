# Dialogue

Creates a dialogue overlay

<Canvas>
  <Story
    name='delete'
    args={{
      title: 'This is the title.',
      message: 'This is the message.',
      buttonText: 'DELETE',
      variant: 'danger',
      open: true,
    }}
  >
    {({ title, message, variant, buttonText, open }) => `
      <v-dialogue
        ${open ? "open" : ""}
        title='${title}'
        message='${message}'
        text='${buttonText}'
        variant='${variant}'
      />
    `}
  </Story>
