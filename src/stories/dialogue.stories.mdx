# Dialogue

Creates a dialogue overlay

<Canvas>
  <Story
    name='delete'
    args={{
      title: 'Delete example.',
      message: 'Are you sure you want to delete this?',
      buttonText: 'DELETE',
      variant: 'danger',
      open: true,
    }}
  >
    {({ title, message, variant, buttonText, open }) => `
      <v-dialogue
        ${open ? "open" : ""}
        title='${title}'
        message='${message}'
      >
        <v-button
          class='ml-4'
          variant='${variant}'
          label='${buttonText}'
        />
      </v-dialogue>
    `}
  </Story>
</Canvas>
